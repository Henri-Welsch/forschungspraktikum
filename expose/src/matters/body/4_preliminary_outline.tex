\section{Vorläufige Gliederung}
\label{sec:vorlaeufige_gliederung}

%% Beginning ===========================


\begin{enumerate}
    \item \textbf{Einleitung}
    \begin{itemize}
        \item Motivation: Relevanz der stätischen Resilienz gegenüber Extremwetterereignissen.
        \item Problemstellung: Vernachlässigung funktionaler Abhängigkeiten (Strom) in bestehenden Erreichbarkeitsmodellen sowie definition der Forschungsfrage.
        \item Zielsetzung: Erweiterung des \textit{RoA-Index} um Kaskadeneffekte und zeitabhängige Notstrompuffer (\textit{Lifetimes}).
    \end{itemize}

    \item \textbf{Grundlagen und Stand der Forschung}
    \begin{itemize}
        \item \textit{RoA}: Definition und technische Umsetzung der physischen Erreichbarkeit nach Kaub et al.\cite{kaub2024}.
        \item Kaskadeneffekte: Modellierung von Stromausfällen und Interdependenzen nach Lickert et al.\cite{lickert2024}.
        \item Datenbasis: Nutzung von OSM zur Repräsentation von Energieinfrastruktur.
    \end{itemize}

    \item \textbf{Konzeptionelles Modell: Funktionaler RoA}
    \begin{itemize}
        \item Definition der funktionalen Verfügbarkeit: Mathematische Verknüpfung von physischem Pfad, Energieversorgung und \textit{Lifetime}.
        \item Topologische Kopplung: Modellierung der Abhängigkeit zwischen POIs und Netzknoten.
        \item Zeitdiskrete Simulation: Beschreibung des Zustandsübergangs von $T+0$ bis $T+n$.
    \end{itemize}

    \item \textbf{Softwaretechnische Implementierung}
    \begin{itemize}
        \item Systemarchitektur: Integration in das bestehende Python-Framework.
        \item Daten-Pipeline: Extraktion von \texttt{power}-Tags via OSM und Aufbau des Stromgraphen.
        \item Algorithmus: Modifikation der Routing-Logik zur dynamischen Filterung funktionaler Zielknoten.
    \end{itemize}

    \item \textbf{Evaluierung und Ergebnisanalyse}
    \begin{itemize}
        \item Case Study: Anwendung auf das Hochwasserszenario in Trier.
        \item Vergleich: Quantifizierung des „funktionalen Gaps“ zwischen physischer und funktionaler RoA\@.
        \item Sensitivitätsanalyse: Untersuchung der Auswirkungen variierender Notstromkapazitäten auf die Gesamtresilienz.
    \end{itemize}

    \item \textbf{Fazit und Ausblick}
    \begin{itemize}
        \item Zusammenfassung: Bestätigung der Hypothese, dass statische Modelle die Resilienz potenziell überschätzen.
        \item Limitationen und zukünftige Forschung: Integration weiterer Sektoren (z. B.\ Wasser und Gas).
    \end{itemize}
\end{enumerate}
